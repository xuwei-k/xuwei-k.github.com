}

case class ResponseStatus(code: Int, message: String) 
  extends Ordered[ResponseStatus] 
{
  def compare(that: ResponseStatus) = code.compareTo(that.code)

  def line = {
    val buf = new StringBuilder(message.length + 5)
    buf.append(code)
    buf.append(' ')
    buf.append(message)
    buf.toString()
  }
}
