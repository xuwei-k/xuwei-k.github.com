package org.scalamock

class MockConstructorDummy

class MockConstructor[R: ClassManifest](factory: MockFactoryBase, name: Symbol) extends MockFunction(factory, name) {

  override private[scalamock] val failIfUnexpected = false

  override private[scalamock] def canHandle(that: MockFunction) = that match {
      case mc: MockConstructor[_] => mockedClass == mc.mockedClass
      case _ => false
    }
  
  val mockedClass = classManifest[R].erasure
}
