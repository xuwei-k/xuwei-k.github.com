package scalikejdbc

import util.control.Exception._

/**
 * Loan pattern implementation
 */
object LoanPattern {

  type Closable = { def close() }

  def using[R <: Closable, A](resource: R)(f: R => A): A = {
    try {
      f(resource)
    } finally {
      ignoring(classOf[Throwable]) apply {
        resource.close()
      }
    }
  }

}
