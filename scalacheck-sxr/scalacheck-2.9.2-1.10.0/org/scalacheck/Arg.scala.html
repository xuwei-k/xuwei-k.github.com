package org.scalacheck

case class Arg[+T](
  label: String,
  arg: T,
  shrinks: Int,
  origArg: T
)(implicit prettyPrinter: T => Pretty) {
  lazy val prettyArg: Pretty = prettyPrinter(arg)
  lazy val prettyOrigArg: Pretty = prettyPrinter(origArg)
}
