package scala

/**
 * An annotation on methods that forbids the compiler to inline the
 * method, no matter how safe the inlining appears to be.
 *
 * @author Lex Spoon
 * @version 1.0, 2007-5-21
 * @since 2.5
 */
class noinline extends scala.annotation.StaticAnnotation
